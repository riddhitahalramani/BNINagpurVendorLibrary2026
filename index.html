var SHEET_ID = "YOUR_SHEET_ID"; function doGet(e) { var action = e.parameter.action || "selections"; var ss = SpreadsheetApp.openById(SHEET_ID); if (action === "companies") { var sh = ss.getSheetByName("Companies"); if (!sh || sh.getLastRow() <= 1) return json({companies:[]}); var data = sh.getRange(2,1,sh.getLastRow()-1,2).getValues(); return json({companies: data.filter(function(r){return r[0];}) .map(function(r){return {name:r[0].toString().trim(),cat:r[1].toString().trim()};})}); } var sh = ss.getSheetByName("Selections"); if (!sh || sh.getLastRow() <= 1) return json({rows:[]}); return json({rows: ss.getSheetByName("Selections") .getRange(2,1,sh.getLastRow()-1,5).getValues()}); } function doPost(e) { var ss = SpreadsheetApp.openById(SHEET_ID); var sh = ss.getSheetByName("Selections") || ss.insertSheet("Selections"); if (sh.getLastRow()===0) sh.appendRow(["Timestamp","Sponsor","Company","Representative","Category"]); JSON.parse(e.postData.contents).rows.forEach(function(r){sh.appendRow(r);}); return json({status:"ok"}); } function json(o){ return ContentService.createTextOutput(JSON.stringify(o)) .setMimeType(ContentService.MimeType.JSON); } // Run seedCompanies() ONCE manually to populate the Companies sheet: function seedCompanies() { /* see the apps-script.js file */ }
